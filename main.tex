\section*{Example 1}
\pgfplotstableread[col sep=comma]{test_data.csv}\datatable
\begin{center}
\begin{tikzpicture}
\begin{axis}[
    axis y line*=none,
    axis x line*=bottom,
    separate axis lines,
    every outer y axis line/.append style=
        {-stealth,white},
    y dir=reverse,
    axis y line shift=10pt,
    axis x line shift=5pt,
    ytick style={color=white},
    xlabel={Value},
    ytick=data,
    yticklabels from table={\datatable}{coef_name},
    ylabel={Coefficient},
    xtick align=outside 
    ]
    \draw [dashed] (axis cs: 0,\pgfkeysvalueof{/pgfplots/ymin}) -- 
                   (axis cs: 0,\pgfkeysvalueof{/pgfplots/ymax});
    \addplot+ [error bars/.cd,
                x explicit,
                x dir=both,
            ] table [
                only marks,
                x=beta_1,
                y expr=\coordindex,
                x error expr=\thisrow{SE_1}
            ] {\datatable};
\end{axis}
\end{tikzpicture}
\end{center}
